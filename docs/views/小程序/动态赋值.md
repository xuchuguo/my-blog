### 动态赋值
1. String & Number
    ```
    <view>{{msg}}</view>
    <button bindtap="change">点我修改信息</button>

    data{
        msg: '我是一条信息'
    },
    change(){
        this.setData({str: this.msg.split('').reverse().join('')})
        // this.setData({str: 123123})  Number
        
    }
    ```

2. Array
    ```
    <view wx:for='{{array}}'>{{item.id}}</view>
    <button bindtap="change">点我+1</button>

    data{
        array: [
            {
                id:1,
                text:1
            },
            {
                id:2,
                text:2
            },
        ]
    },
    change(){
        let arr = this.data.array
        let obj = {id:arr[length-1]+1text:[arr[length-1]+1]}
        arr = arr.concat(obj)
        this.setData({array: arr})

        let arr = this.data.array
        let obj = {id:arr[arr.length-1].id+1,text:[arr[arr.length-1].text+1]}
        arr = arr.concat(obj)
        this.setData({array: arr})
    }
    ```
